# developer-ui

## Install

```bash
// with npm
npm i --save developer-ui

// with yarn
yarn add developer-ui
```

## Usage

```jsx
import { SocketProvider } from 'developer-ui';

...
<SocketProvider apiUrl={...url}>
  <App />
</SocketProvider>
...

import { TaskViewer } from 'developer-ui';

...
<TaskViewer />
...

```
